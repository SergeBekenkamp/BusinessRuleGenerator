-- TRIGGER FOR ATTRIBUTE COMPARE RULE

CREATE OR REPLACE TRIGGER <<trigger_name>>
BEFORE <<trigger_event>>
OF <<column_name>> ON <<entity_name>>
FOR EACH ROW
declare
  l_passed boolean := true;
  l_error_stack varchar2(4000);
BEGIN
  l_passed := :NEW.<<column_name>> <<operator>> '<<value1>>';
  if not l_passed then
    l_error_stack := l_error_stack || 'The value of <<column_name>> is not <<operator>> <<value1>>';
    raise_application_error(-20000, l_error_stack);
  end if;  
END;
